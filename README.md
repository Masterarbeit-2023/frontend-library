```bash
npm install --save frontend-library-test
```

```tsx
import React, { Component } from 'react'

import MyComponent from 'frontend-library-test'
import 'frontend-library-test/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
